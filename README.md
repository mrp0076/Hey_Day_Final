# Hey_Day_Final



```ruby

#!/bin/bash

#Prompt to ask and receive your name
echo "Hi, What's your name?"

read Name


#IF/THEN statement: if no name then asks to input name otherwise, echo name with name tag
#{Name^} capitalizes first letter of the word
if

[ -z "$Name" ]

then
 echo "Oops, pls input a name"

else
echo "Hey ${Name^}!!"
echo "Here's your name tag"

echo "--------------------"
echo "                   "
echo "MY NAME IS    "
echo "                  "
echo "                  "
echo "${Name^}       "
echo "                  "
echo "--------------------"
fi

```
